def read(bytes = nil, output_buffer = nil)
  data = bytes ? nil : ''
  output_buffer ? output_buffer.replace(data || '') : data
end

def write(chunk)
  @block.call(chunk)
  chunk.bytesize.tap { |chunk_size| @size += chunk_size }
end
