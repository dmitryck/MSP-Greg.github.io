def method_missing(method_name, *args, &block)
  if respond_to?(method_name)
    Batch.validate_batch_args!(args)
    batches.each do |batch|
      batch.send(method_name, *args, &block)
    end
  else
    super
  end
end

  end

def first(count = 1)
  if count == 1
    limit(1).to_a.first
  else
    Batch.new(resource_class, limit(count).to_a)
  end
end
