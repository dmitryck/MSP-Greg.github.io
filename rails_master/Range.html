def sum(identity = nil)
  if block_given? || !(first.is_a?(Integer) && last.is_a?(Integer))
    super
  else
    actual_last = exclude_end? ? (last - 1) : last
    if actual_last >= first
      sum = identity || 0
      sum + (actual_last - first + 1) * (actual_last + first) / 2
    else
      identity || 0
    end
  end
end
