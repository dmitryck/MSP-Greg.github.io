def method_missing(method, *args)
  if @helpers.respond_to?(method)
    self.class.class_eval <<-RUBY, __FILE__, __LINE__ + 1
      def #{method}(*args)
        options = args.extract_options!
        args << url_options.merge((options || {}).symbolize_keys)
        @helpers.#{method}(*args)
      end
    RUBY
    send(method, *args)
  else
    super
  end
end
