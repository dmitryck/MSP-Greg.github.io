def register
  Capybara.register_driver @name do |app|
    Capybara::Selenium::Driver.new(app, browser: @browser).tap do |driver|
      driver.browser.manage.window.size = Selenium::WebDriver::Dimension.new(*@screen_size)
    end
  end
end
