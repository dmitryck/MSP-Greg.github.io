def initialize(separator, matchers)
  @separator = separator
  @matchers  = matchers
  @separator_re = "([^#{separator}]+)"
  super()
end

def visit_SYMBOL(node)
  node = node.to_sym

  return @separator_re unless @matchers.key?(node)

  re = @matchers[node]
  "(#{re})"
end
