  until token = scan || @ss.eos?; end
  token

def scan
  case
    # /
  when @ss.skip(/\//)
    [:SLASH, "/"]
  when @ss.skip(/\(/)
    [:LPAREN, "("]
  when @ss.skip(/\)/)
    [:RPAREN, ")"]
  when @ss.skip(/\|/)
    [:OR, "|"]
  when @ss.skip(/\./)
    [:DOT, "."]
  when text = @ss.scan(/:\w+/)
    [:SYMBOL, text]
  when text = @ss.scan(/\*\w+/)
    [:STAR, text]
  when text = @ss.scan(/(?:[\w%\-~!$&'*,;=@]|\\[:()])/)
    text.tr! "\\", ""
    [:LITERAL, text]
    # any char
  when text = @ss.scan(/./)
    [:LITERAL, text]
  end
end
