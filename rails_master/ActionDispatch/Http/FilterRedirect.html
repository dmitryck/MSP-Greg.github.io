def location_filter_match?
  location_filters.any? do |filter|
    if String === filter
      location.include?(filter)
    elsif Regexp === filter
      location =~ filter
    end
  end
end

def location_filters
  if request
    request.get_header("action_dispatch.redirect_filter") || []
  else
    []
  end
end
