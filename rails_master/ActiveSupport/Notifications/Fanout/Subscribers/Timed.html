def finish(name, id, payload)
  timestack = Thread.current[:_timestack]
  started = timestack.pop
  @delegate.call(name, started, Time.now, id, payload)
end

def start(name, id, payload)
  timestack = Thread.current[:_timestack] ||= []
  timestack.push Time.now
