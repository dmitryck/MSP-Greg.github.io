def self.new(pattern, listener)
  if listener.respond_to?(:start) && listener.respond_to?(:finish)
    subscriber = Evented.new pattern, listener
  else
    subscriber = Timed.new pattern, listener
  end

  unless pattern
    AllMessages.new(subscriber)
  else
    subscriber
  end
end
