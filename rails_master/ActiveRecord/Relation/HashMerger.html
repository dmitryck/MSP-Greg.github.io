def other
  other = Relation.create(relation.klass, relation.table, relation.predicate_builder)
  hash.each { |k, v|
    if k == :joins
      if Hash === v
        other.joins!(v)
      else
        other.joins!(*v)
      end
    elsif k == :select
      other._select!(v)
    else
      other.send("#{k}!", v)
    end
  }
  other
end
