def ids
  case value
  when Relation
    value.select(primary_key)
  when Array
    value.map { |v| convert_to_id(v) }
  else
    convert_to_id(value)
  end
end

def polymorphic_base_class_from_value
  case value
  when Relation
    value.klass.base_class
  when Base
    value.class.base_class
  end
end
