def initialize(log_file=nil, level=nil)
  @level = level || INFO
  case log_file
  when String
    @log = open(log_file, "a+")
    @log.sync = true
    @opened = true
  when NilClass
    @log = $stderr
  else
    @log = log_file  # requires "<<". (see BasicLog#log)
  end
end

def format(arg)
  if arg.is_a?(Exception)
    "#{arg.class}: #{arg.message}\n\t" <<
    arg.backtrace.join("\n\t") << "\n"
  elsif arg.respond_to?(:to_str)
    arg.to_str
  else
    arg.inspect
  end
end

def log(level, data)
  if @log && level <= @level
    data += "\n" if /\n\Z/ !~ data
    @log << data
  end
end
