def on_heredoc_beg(tok)
  @stack.push @buf
  buf = []
  @buf << buf
  @buf = buf
  @buf.push Elem.new([lineno(), column()], __callee__, tok)
end

def on_heredoc_dedent(v, w)
  @buf.last.each do |e|
    if e.event == :on_tstring_content
      if (n = dedent_string(e.tok, w)) > 0
        e.pos[1] += n
      end
    end
  end
  v
end
